## Install

```bash
npm install --save use-scroll-detect
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'use-scroll-detect'
import 'use-scroll-detect/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
